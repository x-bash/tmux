___x_cmd_os_loadavg___get () 
{ 
    ___x_cmd_os_name_;
    case "$___X_CMD_OS_NAME_" in 
        darwin | freebsd)
            sysctl -q -n vm.loadavg | cut -d' ' -f2-4
        ;;
        linux | win)
            cut -d' ' -f1-3 < /proc/loadavg
        ;;
        openbsd)
            sysctl -q -n vm.loadavg | cut -d' ' -f1-3
        ;;
    esac
}
